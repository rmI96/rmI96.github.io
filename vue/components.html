Передача параметрів (Props)

<component-child title="Text" />
<component-child :title="var" />
<component-child :title="var + ' text ' + var" />
<component-child :is-title="false" />
<component-child :number-title="28" />




export default {
  props: ["propA", "propB"]
}




export default {
  props: {
    // Basic type check
    propA: Number,
    // Multiple possible types
    propB: [String, Number],
    // Required string
    propC: {
      type: String,
      required: true
    },
    // Number with a default value
    propD: {
      type: Number,
      default: 100
    },
    // Object with a default value
    propE: {
      type: Object,
      default(rawProps) {
        return { message: 'hello' }
      }
    },
    // Custom validator function
    propF: {
      validator(value) {
        return ['success', 'warning', 'danger'].includes(value)
      }
    },
    // Function with a default value
    propG: {
      type: Function,
      default() {
        return 'Default function'
      }
    }
  }
}

<template>

V-model

<component-child
    v-model:first-text="first"
    v-model:second-text="second"
>




<script>
export default {
    props: {
        firstText: String,
        secondText: String
    },
    emits: ['update:firstText', 'update:secondText']
}
</script>

<template>
  <input
      type="text"
      :value="firstText"
      @input="$emit('update:firstText', $event.target.value)"
  />
  <input
      type="text"
      :value="secondText"
      @input="$emit('update:secondText', $event.target.value)"
  />
</template>

Provide / inject

export default {
  data() {
    return {
      message: 'Text'
    }
  },
  provide() {
    return {
      message: this.message
    }
  }
}

Reactivity


import { computed } from 'vue'
export default {
  data() {
    return {
      message: 'Text'
    }
  },
  provide() {
    return {
      message: computed(() => this.message)
    }
  }
}




export default {
  inject: ['message'],
  created() {
    console.log(this.message)
  }
}



export default {
  inject: {
    message: {
      from: 'message',
      default: 'default value'
    },
    user: {
      default: () => ({ name: 'John' })
    }
  }
}
